\begin{align}
    \ket{0}_{\{V(dt)_p\},1} & \otimes \ket{0}_{\{V(dt)_p\},2} \otimes \ldots \\ 
    & \otimes \int \int \int f(\bm{W}) \ket{M,\bm{V}+\bm{W}}_{\{V(dt)_p\},n} dW_xdW_ydW_z\\
    & \qquad \otimes \ket{0}_{\{V(dt)_p\},n+1} \ldots
